\section{Quantitative Replication Results}

\subsection{Ten-seed MNIST confirmation}
\begin{draftpoints}
  \item \draftkey{Central result}opposite published ordering; NDL+IR 0.06395 versus matched CL+IR 0.02367 macro MSE
  \item \draftkey{Intrinsic ratio}2.70$\times$ macro MSE; forgetting 0.01531 versus 0.00742; worse acquisition-sensitive quality and retention
  \item \draftkey{Replay-independent pattern}NDL below matched CL with original-data replay and with no replay; not solely an IR-generator failure
  \item \draftkey{No-replay nuance}lower NDL forgetting; worse macro and foreground MSE; stability--plasticity trade-off, not dominance
\end{draftpoints}

\subsection{Claim-direction ratios}
\begin{draftpoints}
  \item \draftkey{Reading rule}ratio $<1$ favors NDL; all macro-MSE ratios $>1$
  \item \draftkey{Macro-MSE ratios}original data 3.14; intrinsic replay 2.70; no replay 1.91
  \item \draftkey{Forgetting exception}no-replay NDL advantage; simultaneous total-reconstruction deficit
\end{draftpoints}

\subsection{Training resources}
\begin{draftpoints}
  \item \draftkey{Wall time}matched conventional AE 104--110 s; NDL 158--715 s; controlled seed 42
  \item \draftkey{IR overhead}6.88$\times$ slower; 2.31$\times$ peak live CUDA allocation versus same-size AE
  \item \draftkey{Efficiency verdict}additional computation without lower reconstruction error; stronger negative endpoint conclusion
\end{draftpoints}
